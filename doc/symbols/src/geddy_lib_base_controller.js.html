150   /**
151     @name BaseController.formatters.json
152     @static
153     @function
154     @description JSON-formats an object, by looking for toJson/toJSON methds
155     defined on it, and then falling back to JSON.stringify.
156    */
157   this.json = function (content) {
158     var toJson = content.toJson || content.toJSON;
159     if (typeof toJson == 'function') {
160       return toJson.call(content);
161     }
162     return JSON.stringify(content);
163   };
164   /**
165     @name BaseController.formatters.js
166     @static
167     @function
168     @description Formats an object as JSONP, by looking for toJson/toJSON methds
169     defined on it, and then falling back to JSON.stringify, and wrapping it in a
170     callback.
171    */
172   this.js = function (content, controller) {
173     var params = controller.params;
174     if (!params.callback) {
175       err = new errors.InternalServerError('JSONP callback not defined.');
176       controller.error(err);
177     }
178     return params.callback + '(' + JSON.stringify(content) + ');';
179   };
180   /**
181     @name BaseController.formatters.txt
182     @static
183     @function
184     @description Formats an object as plaintext, by looking for a toString
185     defined on it, and then falling back to JSON.stringify.
186    */
187   this.txt = function (content) {
188     if (typeof content.toString == 'function') {
189       return content.toString();
190     }
191     return JSON.stringify(content);
192   };

209   var _addFilter = function (phase, filter, opts) {
210         var obj = {name: filter};
211         obj.except = opts.except;
212         obj.only = opts.only;
213         this[phase + 'Filters'].push(obj);
214       }
215 
216     , _execFilters = function (action, phase, callback) {
217         var _this = this;
218         var filters = this[phase + 'Filters'];
219         var filter;
220         var name;
221         var hook;
222         var list = [];
223         var applyFilter;
224         for (var i = 0; i < filters.length; i++) {
225           filter = filters[i];
226           applyFilter = true;
227           if (filter.only && filter.only != action) {
228             applyFilter = false;
229           }
230           if (filter.except && filter.except == action) {
231             applyFilter = false;
232           }
233           if (applyFilter) {
234             name = filter.name;
235             hook = geddy.hooks.collection[name];
236             hook.args = [_this];
237             list.push(hook);
238           }
239         }
240         var chain = new geddy.async.AsyncChain(list);
241         chain.last = callback;
242         chain.run();
243       }
244 
245     , _negotiateContent = function (frmt) {
246         var format
247           , contentType
248           , types = []
249           , match
250           , params = this.params
251           , err
252           , accepts = this.request.headers.accept
253           , accept
254           , pat
255           , wildcard = false;
256 
257         // If the client provides an Accept header, split on comma
258         // Some user-agents may include whitespace with the comma
259         if (accepts) {
260           accepts = accepts.split(/\s*,\s*/);
261         }
262         // If the client doesn't provide an Accept header, assume
263         // it's happy with anything
264         else {
265           accepts = ['*/*'];
266         }
267 
268         if (frmt) {
269           types = [frmt];
270         }
271         else if (params.format) {
272           var f = params.format;
273           // See if we can actually respond with this format,
274           // i.e., that this one is in the list
275           if (f && ('|' + this.respondsWith.join('|') + '|').indexOf(
276               '|' + f + '|') > -1) {
277             types = [f];
278           }
279         }
280         else {
281           types = this.respondsWith;
282         }
283 
284         // Okay, we have some format-types, let's see if anything matches
285         if (types.length) {
286           for (var i = 0, ii = accepts.length; i < ii; i++) {
287             // Ignore quality factors for now
288             accept = accepts[i].split(';')[0];
289             if (accept == '*/*') {
290               wildcard = true;
291               break;
292             }
293           }
294 
295           // If agent accepts anything, respond with the controller's first choice
296           if (wildcard) {
297             var t = types[0];
298             format = t;
299             contentType = response.formatsPreferred[t];
300             if (!contentType) {
301               this.throwUndefinedFormatError();
302             }
303           }
304           // Otherwise look through the acceptable formats and see if
305           // Geddy knows about any of them.
306           else {
307             for (var i = 0, ii = types.length; i < ii; i++) {
308               pat = response.formatPatterns[types[i]];
309               if (pat) {
310                 for (var j = 0, jj = accepts.length; j < jj; j++) {
311                   match = accepts[j].match(pat);
312                   if (match) {
313                     format = types[i];
314                     contentType = match;
315                     break;
316                   }
317                 }
318               }
319               // If respondsWith contains some random format that Geddy doesn't know
320               // TODO Make it easy for devs to add new formats
321               else {
322                 this.throwUndefinedFormatError();
323               }
324               // Don't look at any more formats if there's a match
325               if (match) {
326                 break;
327               }
328             }
329           }
330         }
331         return {format: format, contentType: contentType};
332       };

336   /**
337    * Primary entry point for calling the action on a controller
338    * Wraps the action so befores and afters can be run
339    */
340   this._handleAction = function (action) {
341     var _this = this;
342     // Wrap the actual action-handling in a callback to use as the 'last'
343     // method in the async chain of before-filters
344     var callback = function () {
345       _this[action].apply(_this, [_this.request, _this.response, _this.params]);
346     };
347     _execFilters.apply(this, [action, 'before', callback]);
348   };

386   /**
387     @name BaseController#redirect
388     @public
389     @function
390     @description Sends a 302 redirect to the client, based on either a
391     simple string-URL, or a controller/action/format combination.
392     @param target {String|Object} Either an URL, or an object literal containing
393     controller/action/format attributes to base the redirect on.
394    */
395   this.redirect = function (target) {
396     var url;
397     if (typeof target == 'string') {
398       url = target;
399     }
400     else {
401       var contr = target.controller || this.name;
402       var act = target.action;
403       var ext = target.format || this.params.format;
404       var id = target.id;
405       contr = geddy.string.decamelize(contr);
406       url = '/' + contr;
407       url += act ? '/' + act : '';
408       url += id ? '/' + id : '';
409       if (ext) {
410         url += '.' + ext;
411       }
412     }
413     var r = new response.Response(this.response);
414     var headers = {'Location': url};
415     if (this.cookies) {
416       headers['Set-Cookie'] = this.cookies.serialize();
417     }
418     this.session.close(function () {
419       r.send('', 302, headers);
420     });
421   };

449   /**
450     @name BaseController#respond
451     @public
452     @function
453     @description Performs content-negotiation, and renders a response.
454     @param content {Object|String} The content to use in the response.
455     @param [format] {String} The desired format for the response.
456    */
457   this.respond = function (content, format) {
458     var c = content
459       , r
460       , headers
461     // format and contentType are set at the same time
462     // format may not be passed
463       , negotiated = _negotiateContent.call(this, format);
464 
465     this.format = negotiated.format;
466     this.contentType = negotiated.contentType;
467 
468     if (!this.contentType) {
469       var err = new errors.NotAcceptableError('Not an acceptable media type.');
470       this.error(err);
471     }
472 
473     // If content needs formatting
474     if (typeof c != 'string') {
475       if (this.format) {
476         c = BaseController.formatters[format](data, this);
477       }
478       // If we couldn't perform content-negotiaton successfully, bail
479       // with error
480       else {
481         this.throwUndefinedFormatError();
482         return;
483       }
484     }
485 
486     this.content = c;
487 
488     r = new response.Response(this.response);
489     headers = {'Content-Type': this.contentType};
490     if (this.cookies) {
491       headers['Set-Cookie'] = this.cookies.serialize();
492     }
493     if (this.session) {
494       this.session.close(function () {
495         r.send(c, 200, headers);
496       });
497     }
498     else {
499       r.send(c, 200, headers);
500     }
501   };
502 
503   this.render = this.respond;

506   this.throwUndefinedFormatError = function () {
507     err = new errors.InternalServerError('Format not defined in response.formats.');
508     this.error(err);
